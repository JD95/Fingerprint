// basic.vert
#version 430 core
layout(location = 0) in vec4 vPosition;
layout(location = 1) in vec4 vColor;
uniform mat4 modelMatrix;
out vec4 color;
void main() {
color = vColor;
gl_Position = modelMatrix * vPosition;
}
// basic.frag
#version 430 core
uniform sampler2D tex;
in vec4 color;
out vec4 fColor;
void main() {
fColor = texture(tex, vec2(color[0],color[1]));
}
